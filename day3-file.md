###### 创建文件

touch 文件名        # 创建空文件/更新文件时间戳
echo "内容" > 文件名  # 覆盖写入内容
echo "内容" >> 文件名 # 追加内容到文件



###### 查看文件

cat 文件名          # 显示文件全部内容（适合短文件）
less 文件名         # 分页查看（上下箭头翻页，/搜索，q退出）
more 文件名         # 分页查看（只能向下翻页）
head 文件名         # 默认显示前10行
head -n 5 文件名    # 显示前5行
tail 文件名         # 默认显示后10行
tail -n 5 文件名    # 显示后5行
tail -f 文件名      # 实时监控文件变化（常用于日志）



###### 复制/移动/删除

cp 源文件 目标路径   # 复制文件
cp -r 源目录 目标路径 # 递归复制目录
mv 旧名 新名        # 重命名/移动文件/目录
mv移动文件夹时，只会移动文件夹本身，不会递归移动里面的内容
